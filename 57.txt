from sympy import isprime, factorint
import math

def count_points(A, B, p):
    count = 1
    for x in range(p):
        rhs = (pow(x,3) + A*x + B) % p
        for y in range(p):
            if (y*y) % p == rhs:
                count += 1
    return count

def ap(A, B, p):
    return p + 1 - count_points(A, B, p)

primes = [p for p in range(3, 300) if isprime(p)]

def bad_primes(A, B):
    disc = 4*A**3 + 27*B**2
    if disc == 0: return None
    return set(factorint(abs(disc)).keys())
