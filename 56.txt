from sympy import isprime, factorint
import math

def count_points(A, B, p):
    count = 1
    for x in range(p):
        rhs = (pow(x,3) + A*x + B) % p
        for y in range(p):
            if (y*y) % p == rhs:
                count += 1
    return count

def ap(A, B, p):
    return p + 1 - count_points(A, B, p)

primes_list = [p for p in range(3, 300) if isprime(p)]

def features(A, B, n=60):
    disc = 4*A**3 + 27*B**2
    if disc == 0: return None
    bad = set(factorint(abs(disc)).keys())
    good = [p for p in primes_list[:n] if p not in bad]
    if len(good) < 15: return None
    aps = [ap(A,B,p) for p in good]
    def zr(ell):
        if ell in bad: return 0
        return sum(1 for a in aps if a%ell==0)/len(aps)
    pts = [(x, int(math.isqrt(x**3+A*x+B)))
           for x in range(-15,30)
           if x**3+A*x+B>=0
           and int(math.isqrt(x**3+A*x+B))**2==x**3+A*x+B]
    disc_f = factorint(abs(disc))
    return [
        zr(2),
        zr(3),
        zr(5),
        math.log(abs(disc)),
        len(disc_f),
        max(disc_f.values()),
        len(pts),
    ]

def mean_vec(rows):
    return [sum(col)/len(col) for col in zip(*rows)]

def cov_matrix(rows, idx):
    # idx で指定した3特徴量の 3×3 共分散行列
    n = len(rows)
    sub = [[row[i] for i in idx] for row in rows]
    mu = [sum(col)/n for col in zip(*sub)]
    C = [[0]*3 for _ in range(3)]
    for row in sub:
        d = [row[i]-mu[i] for i in range(3)]
        for i in range(3):
            for j in range(3):
                C[i][j] += d[i]*d[j]/n
    return C

def eigenvalues_3x3(M):
    # 特性多項式の係数
    a = M[0][0]; b = M[0][1]; c = M[0][2]
    d = M[1][0]; e = M[1][1]; f = M[1][2]
    g = M[2][0]; h = M[2][1]; k = M[2][2]
    tr = a+e+k
    det = (a*(e*k-f*h) - b*(d*k-f*g) + c*(d*h-e*g))
    # 固有値の積、和、ペア和
    p2 = a*e+e*k+a*k - b*d - f*h - c*g
    # x³ - tr x² + p2 x - det = 0
    # 数値解
    import cmath
    p = p2 - tr**2/3
    q = 2*tr**3/27 - tr*p2/3 + det
    disc_e = -(4*p**3 + 27*q**2)
    if abs(p) < 1e-10:
        return [tr/3, tr/3, tr/3]
    D = (q/2)**2 + (p/3)**3
    sqrt_D = cmath.sqrt(D)
    u = (-q/2 + sqrt_D)**(1/3)
    v = (-q/2 - sqrt_D)**(1/3)
    roots = []
    for k in range(3):
        w = cmath.exp(2j*math.pi*k/3)
        r = w*u + v/w if abs(w) > 1e-10 else u+v
        roots.append((r + tr/3).real)
    return sorted(roots)
